## Option
Option是一个最多包含一个元素的List
### Demo1: 定义Option数据类型
```Scala
sealed trait Option[+A] {
    // 如果Option不为None, 对其应用f
    def map[B](f: A => B): Option[B] =
        this match {
            case None => None
            case Some(a) => Some(f(a))
        }

    def flatMap[B](f: A => Option[B]): Option[B] =
        this match {
            case None => None
            case Some(a) => f(a)
        }

    def flatMapViaMap[B](f: A => Option[B]): Option[B] =
        map(f) getOrElse None


    // => B 表示参数类型是B, 但不是立即求值
    // B >:A 表示类型参数必须是A的父类型
    def getOrElse[B >: A](default: => B): B =
        this match {
            case None => default
            case Some(a) => a
        }

    // 如果本Option为空, 就返回ob
    def orElse[B >: A](ob: => Option[B]): Option[B] =
        this match {
            case None => ob
            case _ => this
        }

    def orElseViaMap[B >: A](ob: => Option[B]): Option[B] =
        this map (Some(_)) getOrElse ob

    // 如果值不满足f, 转换Some为None 
    // 后面一个匹配了两种情况: 1.本身为None 2.为Some但不满足f条件
    def filter(f: A => Boolean): Option[A] =
        this match {
        case Some(a) if f(a) => this
        case _ => None
        }

    def filterViaFlatMap(f: A => Boolean): Option[A] =
        flatMap(a => if (f(a)) Some(a) else None)

}
case class Some[+A](get: A) extends Option[A]
case object None extends Option[Nothing]


def mean(xs: Seq[Double]): Option[Double] =
    if (xs.isEmpty) None
    else Some(xs.sum / xs.length)

```

### Demo3: 使用二元函数组合两个Option: map2
```Scala
def map2[A,B,C](a: Option[A], b: Option[B])(f: (A,B) => C): Option[C] =
    (a, b) match {
        case (Some(a), Some(b)) => Some(f(a, b))
        case _ => None
    }

 
def map2[A,B,C](a: Option[A], b: Option[B])(f: (A,B) => C): Option[C] =
    a flatMap (aa => b map (bb => f(aa, bb)))
    
```

### Demo4: sequence: 将Option列表结合为一个Option
```Scala
def sequence[A](a: List[Option[A]]): Option[List[A]] =
    a match {
        case Nil => Some(Nil)
        case h::t => h flatMap (hh => sequence(t) map (hh :: _))
    }

def sequence[A](a: List[Option[A]]): Option[List[A]] =
    a.foldRight[Option[List[A]]](Some(Nil))((x, y) => map2(x, y)(_ :: _))
```

## Either

### Demo6: Either定义
```Scala
trait Either[+E, +A] {
    def map[B](f: A => B): Either[E, B] =
        this match {
            case Left(e) => Left(e)
            case Right(a) => Right(f(a))
        }

    def flatMap[EE >: E, B](f: A => Either[EE, B]): Either[EE, B] = 
        this match {
            case Left(e) => Left(e)
            case Right(a) => f(a)
        }

    def orElse[EE >: E, B >: A](b: => Either[EE, B]): Either[EE, B] = 
        this match {
            case Left(_) => b
            case Right(a) => Right(a)
        }

    def map2[EE >: E, B, C](b: Either[EE, B])(f: (A,B) => C): Either[EE, C] =
        for {
            a <- this
            b1 <- b
        } yield f(a, b1)
}
case class Left[+E](value: E) extends Either[E, Nothing]
case class Right[+A](value: A) extends Either[Nothing, A]
```
